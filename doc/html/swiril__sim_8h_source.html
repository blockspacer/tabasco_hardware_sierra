35 /* The SIM's EF File Header size is variable, but
36  * there's no harm in asking for more bytes than
37  * are available. The SIM will never reply with
38  * more than the header size. So the value below
39  * is designed to ensure we always get the maximum
40  * number of bytes for the header for proper parsing
41  */
42 #define SIM_HEADER_REQ_SIZE 64
43 
44 /* The SIM's EF File Header size needs to be zero
45  * for QMI RIL else it doesn't work to get
46  * SIM contacts.
47  */
48 #define SIM_HEADER_REQ_SIZE_QMI 0
