158 DESCRIPTION
159  Masks the bits in data at the given offset for given number of width bits.

161 #define MASK(width, offset, data) \

169  (((width) == bitsize(data)) ? (data) : \
170  ((((copymask(data) << (bitsize(data) - ((width) % bitsize(data)))) \
171  & copymask(data)) >> (offset)) & (data))) \
172  \
173  /*lint +e701 */ \

175  /*lint +e572 */ \
